## Endpoint

```
GET /api/routines
```

Retrieve a paginated list of workout routines from the BodyWorks database. Each routine includes a complete workout plan with daily schedules and workout summaries.

## Query Parameters

<ParamField query="limit" type="number" required>
  The number of routines to return per page. Controls pagination size.
</ParamField>

<ParamField query="page" type="number" required>
  The page number to retrieve. Used for pagination through the routine list.
</ParamField>

## Response

<ResponseField name="totalRoutines" type="number">
  The total number of routines available in the database.
</ResponseField>

<ResponseField name="totalPages" type="number">
  The total number of pages available based on the limit parameter.
</ResponseField>

<ResponseField name="data" type="IRoutine[]">
  An array of routine objects.
  
  <Expandable title="Routine Object">
    <ResponseField name="id" type="number">
      The unique identifier for the routine.
    </ResponseField>
    
    <ResponseField name="id_" type="number">
      Alternative identifier for the routine.
    </ResponseField>
    
    <ResponseField name="category" type="string[]">
      Array of categories this routine belongs to (e.g., ["strength", "mass-building"]).
    </ResponseField>
    
    <ResponseField name="routine" type="object">
      The main routine object containing all routine details.
      
      <Expandable title="Routine Details">
        <ResponseField name="routine_title" type="string">
          The title of the workout routine.
        </ResponseField>
        
        <ResponseField name="routine_description" type="string">
          Detailed description of the routine, including goals and methodology.
        </ResponseField>
        
        <ResponseField name="routine_imageUrl" type="string">
          URL to the routine's featured image.
        </ResponseField>
        
        <ResponseField name="workout_plan" type="object[]">
          Array of daily workout plans.
          
          <Expandable title="Workout Plan">
            <ResponseField name="heading" type="string">
              The day or phase heading (e.g., "Day 1: Chest & Triceps").
            </ResponseField>
            
            <ResponseField name="day_plan" type="string">
              Detailed plan for this day including exercises, sets, and reps.
            </ResponseField>
          </Expandable>
        </ResponseField>
        
        <ResponseField name="workout_summary" type="object">
          Summary of the routine's key characteristics.
          
          <Expandable title="Workout Summary">
            <ResponseField name="MainGoal" type="string">
              The primary goal of the routine (e.g., "Build Muscle", "Lose Fat").
            </ResponseField>
            
            <ResponseField name="WorkoutType" type="string">
              The type of workout (e.g., "Split", "Full Body").
            </ResponseField>
            
            <ResponseField name="TrainingLevel" type="string">
              Target experience level (e.g., "Beginner", "Intermediate", "Advanced").
            </ResponseField>
            
            <ResponseField name="ProgramDuration" type="string">
              Total duration of the program (e.g., "8 weeks", "12 weeks").
            </ResponseField>
            
            <ResponseField name="DaysPerWeek" type="number">
              Number of training days per week.
            </ResponseField>
            
            <ResponseField name="TimePerWorkout" type="string">
              Average time per workout session (e.g., "45-60 minutes").
            </ResponseField>
            
            <ResponseField name="EquipmentRequired" type="string">
              Equipment needed for the routine (e.g., "Full Gym", "Dumbbells Only").
            </ResponseField>
            
            <ResponseField name="TargetGender" type="string">
              Target gender for the routine (e.g., "Male", "Female", "Unisex").
            </ResponseField>
          </Expandable>
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>

## Example Request

<Tabs>
  <Tab title="cURL">
    ```bash
    curl -X GET "https://api.bodyworks.com/api/routines?limit=10&page=1"
    ```
  </Tab>
  
  <Tab title="axios">
    ```javascript
    import axios from 'axios';

    const getRoutines = async (limit = 10, page = 1) => {
      const response = await axios.get('/api/routines', {
        params: {
          limit,
          page
        }
      });
      return response.data;
    };

    // Usage
    const routines = await getRoutines(10, 1);
    ```
  </Tab>
  
  <Tab title="fetch">
    ```javascript
    const getRoutines = async (limit = 10, page = 1) => {
      const response = await fetch(
        `/api/routines?limit=${limit}&page=${page}`
      );
      const data = await response.json();
      return data;
    };

    // Usage
    const routines = await getRoutines(10, 1);
    ```
  </Tab>
</Tabs>
