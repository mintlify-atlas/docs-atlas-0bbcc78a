## Overview

The filter types define the structure for exercise filtering options including body parts, equipment, and target muscles. These types enable type-safe filtering and categorization of exercises.

## IBodyPart

Represents a body part category with visual representation.

```typescript types.d.ts
interface IBodyPart {
  bodyPart: string;
  imageUrl: string;
}
```

### Properties

<ResponseField name="bodyPart" type="string" required>
  Name of the body part (e.g., "chest", "back", "legs", "shoulders")
</ResponseField>

<ResponseField name="imageUrl" type="string" required>
  Image URL representing the body part for visual selection
</ResponseField>

## IBodyPartData

```typescript types.d.ts
interface IBodyPartData {
```

    ```typescript
    import { IBodyPart, IEquipment, ITargetMuscle } from './types';

    interface FilterState {
      bodyParts: IBodyPart[];
      equipment: IEquipment[];
      targetMuscles: ITargetMuscle[];
      selected: {
        bodyPart?: string;
        equipment?: string;
        targetMuscle?: string;
      };
    }

    type FilterAction =
      | { type: 'SET_BODY_PARTS'; payload: IBodyPart[] }
      | { type: 'SET_EQUIPMENT'; payload: IEquipment[] }
      | { type: 'SET_TARGET_MUSCLES'; payload: ITargetMuscle[] }
      | { type: 'SELECT_BODY_PART'; payload: string }
      | { type: 'SELECT_EQUIPMENT'; payload: string }
      | { type: 'SELECT_TARGET_MUSCLE'; payload: string }
      | { type: 'CLEAR_FILTERS' };

    function filterReducer(
      state: FilterState,
      action: FilterAction
    ): FilterState {
      switch (action.type) {
        case 'SET_BODY_PARTS':
          return { ...state, bodyParts: action.payload };
        case 'SET_EQUIPMENT':
          return { ...state, equipment: action.payload };
        case 'SET_TARGET_MUSCLES':
          return { ...state, targetMuscles: action.payload };
        case 'SELECT_BODY_PART':
          return {
            ...state,
            selected: { ...state.selected, bodyPart: action.payload }
          };
        case 'SELECT_EQUIPMENT':
          return {
            ...state,
            selected: { ...state.selected, equipment: action.payload }
          };
        case 'SELECT_TARGET_MUSCLE':
          return {
            ...state,
            selected: { ...state.selected, targetMuscle: action.payload }
          };
        case 'CLEAR_FILTERS':
          return {
            ...state,
            selected: {}
          };
        default:
          return state;
      }
    }

    // Hook for using filters
    import { useReducer, useEffect } from 'react';

    export function useFilters() {
      const [state, dispatch] = useReducer(filterReducer, {
        bodyParts: [],
        equipment: [],
        targetMuscles: [],
        selected: {}
      });

      useEffect(() => {
        async function loadFilters() {
          const [bodyParts, equipment, targetMuscles] = await Promise.all([
            fetch('https://api.bodyworks.io/bodyparts').then(r => r.json()),
            fetch('https://api.bodyworks.io/equipment').then(r => r.json()),
            fetch('https://api.bodyworks.io/targets').then(r => r.json())
          ]);

          dispatch({ type: 'SET_BODY_PARTS', payload: bodyParts.data });
          dispatch({ type: 'SET_EQUIPMENT', payload: equipment.data });
          dispatch({ type: 'SET_TARGET_MUSCLES', payload: targetMuscles.data });
        }

        loadFilters();
      }, []);

      return { state, dispatch };
    }
    ```

## How Filter Types Work Together

The filter types are designed to work together to provide a comprehensive exercise filtering system:

```typescript
import {
  IBodyPartData,
  IEquipmentData,
  ITargetMuscleData,
  IExerciseData,
  IExercise
} from './types';

class ExerciseFilterService {
  private bodyParts: IBodyPartData | null = null;
  private equipment: IEquipmentData | null = null;
  private targetMuscles: ITargetMuscleData | null = null;

  async initialize() {
    [this.bodyParts, this.equipment, this.targetMuscles] = await Promise.all([
      fetch('https://api.bodyworks.io/bodyparts').then(r => r.json()),
      fetch('https://api.bodyworks.io/equipment').then(r => r.json()),
      fetch('https://api.bodyworks.io/targets').then(r => r.json())
    ]);
  }

  getAvailableBodyParts(): string[] {
    return this.bodyParts?.data.map(bp => bp.bodyPart) || [];
  }

  getAvailableEquipment(): string[] {
    return this.equipment?.data.map(eq => eq.equipment) || [];
  }

  getAvailableTargets(): string[] {
    return this.targetMuscles?.data.map(tm => tm.targetMuscle) || [];
  }

  async searchExercises(filters: {
    bodyPart?: string;
    equipment?: string;
    target?: string;
  }): Promise<IExercise[]> {
    const params = new URLSearchParams();
    if (filters.bodyPart) params.append('bodyPart', filters.bodyPart);
    if (filters.equipment) params.append('equipment', filters.equipment);
    if (filters.target) params.append('target', filters.target);

    const response = await fetch(
      `https://api.bodyworks.io/exercises?${params}`
    );
    const data: IExerciseData = await response.json();
    return data.data;
  }

  getFilterImage(type: 'bodyPart' | 'equipment' | 'target', value: string): string | undefined {
    switch (type) {
      case 'bodyPart':
        return this.bodyParts?.data.find(bp => bp.bodyPart === value)?.imageUrl;
      case 'equipment':
        return this.equipment?.data.find(eq => eq.equipment === value)?.imageUrl;
      case 'target':
        return this.targetMuscles?.data.find(tm => tm.targetMuscle === value)?.imageUrl;
    }
  }
}

// Usage
const filterService = new ExerciseFilterService();
await filterService.initialize();

const chestExercises = await filterService.searchExercises({
  bodyPart: 'chest',
  equipment: 'dumbbell'
});
```

## Type Guards

Create type guards to safely validate filter data:

```typescript
function isBodyPart(obj: any): obj is IBodyPart {
  return (
    typeof obj === 'object' &&
    typeof obj.bodyPart === 'string' &&
    typeof obj.imageUrl === 'string'
  );
}

function isEquipment(obj: any): obj is IEquipment {
  return (
    typeof obj === 'object' &&
    typeof obj.equipment === 'string' &&
    typeof obj.imageUrl === 'string'
  );
}

function isTargetMuscle(obj: any): obj is ITargetMuscle {
  return (
    typeof obj === 'object' &&
    typeof obj.targetMuscle === 'string' &&
    typeof obj.imageUrl === 'string'
  );
}
```
