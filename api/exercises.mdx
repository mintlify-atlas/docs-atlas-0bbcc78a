## Endpoint

```
GET /api/exercises
```

Retrieve a paginated list of exercises from the BodyWorks database. This endpoint supports pagination to efficiently load large sets of exercise data.

## Query Parameters

<ParamField query="limit" type="number" default="9">
  The number of exercises to return per page. Controls pagination size.
</ParamField>

<ParamField query="page" type="number" default="1">
  The page number to retrieve. Used for pagination through the exercise list.
</ParamField>

## Response

<ResponseField name="totalExercises" type="number">
  The total number of exercises available in the database.
</ResponseField>

<ResponseField name="totalPages" type="number">
  The total number of pages available based on the limit parameter.
</ResponseField>

<ResponseField name="data" type="IExercise[]">
  An array of exercise objects.
  
  <Expandable title="Exercise Object">
    <ResponseField name="name" type="string">
      The unique identifier name of the exercise.
    </ResponseField>
    
    <ResponseField name="title" type="string">
      The display title of the exercise.
    </ResponseField>
    
    <ResponseField name="target" type="string">
      The primary target muscle or muscle group.
    </ResponseField>
    
    <ResponseField name="muscles_worked" type="string">
      Description of all muscles worked during the exercise.
    </ResponseField>
    
    <ResponseField name="bodyPart" type="string">
      The body part category (e.g., "chest", "legs", "back").
    </ResponseField>
    
    <ResponseField name="equipment" type="string">
      The equipment required for this exercise.
    </ResponseField>
    
    <ResponseField name="id" type="string">
      The unique identifier for the exercise.
    </ResponseField>
    
    <ResponseField name="id_" type="string">
      Alternative identifier for the exercise.
    </ResponseField>
    
    <ResponseField name="blog" type="string">
      URL or content for related blog post about the exercise.
    </ResponseField>
    
    <ResponseField name="images" type="string[]">
      Array of image URLs showing the exercise.
    </ResponseField>
    
    <ResponseField name="gifUrl" type="string">
      URL to an animated GIF demonstrating the exercise.
    </ResponseField>
    
    <ResponseField name="videos" type="string[]">
      Array of video URLs demonstrating the exercise.
    </ResponseField>
    
    <ResponseField name="keywords" type="string[]">
      Array of keywords associated with the exercise for search and filtering.
    </ResponseField>
  </Expandable>
</ResponseField>

## Example Request

<Tabs>
  <Tab title="cURL">
    ```bash
    curl -X GET "https://api.bodyworks.com/api/exercises?limit=10&page=1"
    ```
  </Tab>
  
  <Tab title="axios">
    ```javascript
    import axios from 'axios';

    const getExercises = async (limit = 10, page = 1) => {
      const response = await axios.get('/api/exercises', {
        params: {
          limit,
          page
        }
      });
      return response.data;
    };

    // Usage
    const exercises = await getExercises(10, 1);
    ```
  </Tab>
  
  <Tab title="fetch">
    ```javascript
    const getExercises = async (limit = 10, page = 1) => {
      const response = await fetch(
        `/api/exercises?limit=${limit}&page=${page}`
      );
      const data = await response.json();
      return data;
    };

    // Usage
    const exercises = await getExercises(10, 1);
    ```
  </Tab>
</Tabs>
